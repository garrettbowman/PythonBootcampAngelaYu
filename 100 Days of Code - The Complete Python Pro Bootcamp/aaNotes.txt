Pandas list comprehension

import pandas as pd

# Create a sample DataFrame

data = {'col1': [1, 2, 3], 'col2': ['A', 'B', 'C']}

df = pd.DataFrame(data)

# Example: Create a list of tuples (index, value_from_col1)

new_list = [(index, row['col1']) for index, row in df.iterrows()]

print(new_list)
